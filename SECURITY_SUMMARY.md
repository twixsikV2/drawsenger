## Как использовать

### 1. Импортируйте функции

```typescript
import { sanitizeText, validateEmail, detectSQLInjection } from './lib/security';
import { bruteForceProtection, xssProtection } from './lib/advanced-security';
```

### 2. Используйте в компонентах

```typescript
// Валидируйте входные данные
if (!validateEmail(email)) throw new Error('Неверный email');

// Проверяйте на атаки
if (detectSQLInjection(userInput)) throw new Error('Атака обнаружена');

// Санитизируйте данные
const safe = sanitizeText(userInput);

// Логируйте события
logSecurityEvent('USER_LOGIN', { userId }, 'low');
```
